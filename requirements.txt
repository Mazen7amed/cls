$file = ""C:\Users\hmaze\Desktop\Go Export Final\Assets\image_classification_model.pt""
$chunkSize = 99MB
$i = 0
$reader = [System.IO.File]::OpenRead($file)
while ($reader.Position -lt $reader.Length) {
    $buffer = New-Object byte[] $chunkSize
    $bytesRead = $reader.Read($buffer, 0, $chunkSize)
    $chunk = $buffer[0..($bytesRead-1)]
    [System.IO.File]::WriteAllBytes("$file.part$i", $chunk)
    $i++
}
$reader.Close()
